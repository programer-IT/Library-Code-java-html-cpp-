class Student { private String name; public void setName(String n) { name = n; } public String getName() { return name; } } public class Main { public static void main(String[] args) { Student s = new Student(); s.setName("Alice"); System.out.println(s.getName()); } }

Encapsulation

Encapsulation hides data using private variables; access through getters and setters.

class Animal { void sound() { System.out.println("Animal makes a sound"); } } class Dog extends Animal { void sound() { System.out.println("Dog barks"); } } public class Main { public static void main(String[] args) { Animal a = new Animal(); Dog d = new Dog(); a.sound(); d.sound(); } }
